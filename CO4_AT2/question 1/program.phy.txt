# Warehouse Delivery Route Optimization
# Using Prim's Algorithm (Greedy Strategy)

INF = float('inf')

# Graph represented using adjacency matrix
graph = [
    [0, 2, 4],
    [2, 0, 1],
    [4, 1, 0]
]

V = len(graph)

# Keep track of selected warehouses
selected = [False] * V
selected[0] = True

edges = 0
total_weight = 0

print("Selected Routes:")

# MST contains V-1 edges
while edges < V - 1:

    minimum = INF
    x = 0
    y = 0

    # Find minimum cost edge
    for i in range(V):
        if selected[i]:
            for j in range(V):
                if (not selected[j]) and graph[i][j]:
                    if graph[i][j] < minimum:
                        minimum = graph[i][j]
                        x = i
                        y = j

    print(f"Warehouse {x} --> Warehouse {y} : Distance = {graph[x][y]}")
    total_weight += graph[x][y]

    selected[y] = True
    edges += 1

print("\nTotal Weight of MST =", total_weight)

input("\nPress Enter to Exit...")
